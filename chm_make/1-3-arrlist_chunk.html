arrlist_chunk($arrlist, $key)

对二维数组进行分段 。

$arrlist: 二维数组
$key: Key
【定义】 文件：xiunophp/xiunophp.php 大约 881 行： 
function arrlist_chunk($arrlist, $key) {
	$r = array();
	if(empty($arrlist)) return $r;
	foreach($arrlist as &$arr) {
		!isset($r[$arr[$key]]) AND $r[$arr[$key]] = array();
		$r[$arr[$key]][] = $arr;
	}
	return $r;
}

【用例】 
<?php

$conf = include './conf.php';
include './xiunophp/xiunophp.php';

$arrlist = array(array('a'=>1, 'b'=>2), array('a'=>3, 'b'=>4));
$r = arrlist_chunk($arrlist, 'b');
print_r($r);

/* 结果：
Array

                    [b] => 2
                )

        )

    [4] => Array
        (
            [0] => Array
                (
                    [a] => 3
                    [b] => 4
                )

        )

)
*/

?>
