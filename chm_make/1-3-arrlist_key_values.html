arrlist_key_values($arrlist, $key, $value)

从一个二维数组中取出一个 key=>value 格式的一维数组。

$arrlist: 二维数组
$key: 键
$value: 值
【定义】 文件：xiunophp/xiunophp.php 大约 373 行： 
function arrlist_key_values($arrlist, $key, $value) {
	$return = array();
	if($key) {
		foreach((array)$arrlist as $arr) {
			$return[$arr[$key]] = $arr[$value];
		}
	} else {
		foreach((array)$arrlist as $arr) {
			$return[] = $arr[$value];
		}
	}
	return $return;
}

【用例】 
<?php

$conf = include './conf.php';
include './xiunophp/xiunophp.php';

$arrlist = array(
	array('a'=>3,'b'=>4,'c'=>7),
	array('a'=>2,'b'=>5,'c'=>8),
	array('a'=>1,'b'=>6,'c'=>9),
);
$arrlist = arrlist_key_values($arrlist, 'a', 'b');
print_r($arrlist);

/* 返回：
Array
(
    [3] => 4
    [2] => 5
    [1] => 6
)*/

?>
