获取 URL 的路径部分，区别于 $_SERVER['REQUEST_URI'] 。
【定义】 文件：xiunophp/xiunophp.php 大约 1458 行： 
function http_url_path() {
	$port = $_SERVER['SERVER_PORT'];
	//$portadd = ($port == 80 ? '' : ':'.$port);
	$host = $_SERVER['HTTP_HOST'];  // host 里包含 port
	$path = substr($_SERVER['PHP_SELF'], 0, strrpos($_SERVER['PHP_SELF'], '/'));
	$http = (($port == 443) || (isset($_SERVER['HTTPS']) && strtolower($_SERVER['HTTPS']) != 'off')) ? 'https' : 'http';
	return  "$http://$host$path/";
}

【用例】 
<?php

$conf = include './conf.php';
include './xiunophp/xiunophp.php';

// 假如当前 URL 为：http://bbs.xiuno.com/admin/index.php
$s = http_url_path();
print_r($s); // http://www.xiuno.com/admin/

?>
