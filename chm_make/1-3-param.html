param($key, $defval = '', $safe = TRUE)

获取 $_GET $_POST $_COOKIE 中的参数。

$key：键
$defval: 默认值，设置以后类型会被强制转换
$safe: 是否进行 HTML 转义，默认转义
【定义】 文件：xiunophp/xiunophp.php 大约 171 行： 
/*
	param(1);
	param(1, '');
	param(1, 0);
	param(1, array());
	param(1, array(''));
	param(1, array(0));
*/
function param($key, $defval = '', $safe = TRUE) {
	if(!isset($_REQUEST[$key]) || ($key === 0 && empty($_REQUEST[$key]))) {
		if(is_array($defval)) {
			return array();
		} else {
			return $defval;
		}
	}
	$val = $_REQUEST[$key];
	$val = param_force($val, $defval, $safe);
	return $val;
}

【用例】

$conf = include './conf.php';
include './xiunophp/xiunophp.php';

// http://domain.com/user-login.htm?uid=123

$b = param(1); // "login"
$uid = param('uid', 0); // 123
