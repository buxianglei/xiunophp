function form_radio($name, $arr, $checked = 0) {
	empty($arr) && $arr = array('否', '是');
	$s = '';

	foreach((array)$arr as $k=>$v) {
		$add = $k == $checked ? ' checked="checked"' : '';
		$s .= "   \r\n";
	}
	return $s;
}
