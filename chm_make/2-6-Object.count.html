Object.count(obj)

获取 Object 的元素个数，返回一个数值。
【定义】 文件：xiunoui/xiuno.js 大约 55 行： 
Object.count = function(obj) {
	if(!obj) return 0;
	if(obj.length) return obj.length;
	var n = 0;
	for(k in obj) {
		if(obj.hasOwnProperty(k)) n++;
	}
	return n;
}

【用例】 
var obj = {"a":1, "b":2};
var n = Object.count(obj);
console.log(n); // 2
