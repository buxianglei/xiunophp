floatval()

将一个变量转换为浮点类型。
与 parseInt 不同的是它不会返回 NaN 对象。
【定义】 文件：xiunoui/xiuno.js 大约 117 行： 
global.floatval = function(s) {
    if(!s) return 0;
    if(s.constructor === Array) {
        for(var i=0; i<s.length; i++) {
            s[i] = floatval(s[i]);
        }
        return s;
    }
    var r = parseFloat(s);
    return isNaN(r) ? 0 : r;
}

【用例】 
var i = '12.3 s';
console.log(i);
// 12.3
