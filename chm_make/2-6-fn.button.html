$.fn.button(status)

设置按钮的状态和文字。

status: 
	loading: 置按钮上的文字为 loading-tex 属性值，并且为禁用状态
	disabled: 按钮设置为禁用状态
	enable: 按钮设置为启用状态
	reset: 按钮文字状态复位
	其他：将按钮的文字设置为“其他”
【定义】 文件：xiunoui/xiuno.js 大约 1289 行： 
$.fn.button = function(status) {
	return this.each(function() {
		if(status == 'loading') {
			$(this).prop('disabled', true).addClass('disabled').attr('default-text', $(this).text());
			$(this).html($(this).attr('loading-text'));
		} else if(status == 'disabled') {
			$(this).prop('disabled', true).addClass('disabled');
		} else if(status == 'enable') {
			$(this).prop('disabled', false).removeClass('disabled');
		} else if(status == 'reset') {
			$(this).prop('disabled', false).removeClass('disabled');
			if($(this).attr('default-text')) {
				$(this).text($(this).attr('default-text'));
			}
		} else {
			$(this).text(status);
		}
	});
}

【用例】 
<button loading-text="正在提交..." class="blue big" id="button1">下一步</button>
<script>
$('#button1').button('loading');
</script>
