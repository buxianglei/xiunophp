$.fn.menu_show(menuid, pos, timeout)

将菜单浮动于控件周围，鼠标离开会有一个延迟才消失。

menuid: 菜单ID
pos: 位置
timeout: 几秒后自动消失
【定义】 文件：xiunoui/xiuno.js 大约 1204 行： 
// 将菜单浮动于 this 对象的周围, pos 默认为 8
/*
	1		2		3
	4				6
	7		8		9
*/
$.fn.menu_show = function(menuid, pos, timeout) {
	return this.each(function() {
		var jthis = $(this);
		var jmenu = $(menuid);
		jmenu.css({position: "absolute", top: '-1000px', left : '-1000px', 'z-index':10000}).show();
		if(!pos) pos = 8;
		var offset2 = $.link_div_position(jthis, jmenu, pos, true);
		if(!offset2) return;
		jmenu.css(offset2);
		jmenu.fadeIn('fast');
		if(timeout) jmenu.mouseout_hide(timeout, jthis);
	});
};

【用例】 <input type="text" value="姓名" id="input1"> <div id="menu1" style="width: 100px; height: 100px;">层内内容</div> <script> $('#input1').menu_show('menu1', 6); // 弹出层 </script>
