is_email(s)

判断字符串是否为 Email 格式，xxx@xxx.xxx，返回布尔值。
【定义】 文件：xiunoui/xiuno.js 大约 335 行： 
global.is_email = function(s) {
	var r = /^[\w\-\.]+@[\w\-\.]+(\.\w+)+$/i
	if(!s) {
		return false;
	} else if(!r.test(s)) {
		return false;
	}
	return true;
}

【用例】 
var s = "zhangsan@gmail.com";
var r = is_email(s);
console.log(r);
// true
