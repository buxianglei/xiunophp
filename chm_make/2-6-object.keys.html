Object.keys(obj)

获取 Object 的 keys，返回一个数组。
主要为了兼容 IE89 。
【定义】 文件：xiunoui/xiuno.js 大约 32 行： 
if(!Object.keys) {
	Object.keys = function(o) {
		var arr = [];
		for(var k in o) {
			if(o.hasOwnProperty(k)) arr.push(o[k]);
		}
		return arr;
	}
}

【用例】 
var obj = {"a":1, "b":2};
var arr = Object.keys(obj);
console.log(arr); // ["a", "b"]
