parse_str(str)

解析 URL 参数格式的字符串，返回 json 格式的数据。
【定义】 文件：xiunoui/xiuno.js 大约 385 行： 
global.parse_str = function (str){
	var sep1 = '=';
	var sep2 = '&';
	var arr = str.split(sep2);
	var arr2 = {};
	for(var x=0; x < arr.length; x++){
		var tmp = arr[x].split(sep1);
		arr2[unescape(tmp[0])] = unescape(tmp[1]).replace(/[+]/g, ' ');
	}
	return arr2;
}

【用例】 
var json = parse_str('a=1&b=2&c=3=');
console.log(json);
/*
	{
		"a":1,
		"b":2,
		"c":3,
	}
*/
