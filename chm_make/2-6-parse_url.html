parse_url(url)

解析 URL，返回 json 格式的数据，包含 URL 各种信息。
【定义】 文件：xiunoui/xiuno.js 大约 385 行： 
global.parse_url = function(url) {
	if(url.match(/^(([a-z]+):)\/\//i)) {
		var arr = url.match(/^(([a-z]+):\/\/)?([^\/\?#]+)\/*([^\?#]*)\??([^#]*)#?(\w*)$/i);
		if(!arr) return null;
		var r = {
		    'schema': arr[2],
		    'host': arr[3],
		    'path': arr[4],
		    'query': arr[5],
		    'anchor': arr[6],
		    'requesturi': arr[4] + (arr[5] ? '?'+arr[5] : '') + (arr[6] ? '#'+arr[6] : '')
		};
		console.log(r);
		return r;
	} else {
		
		var arr = url.match(/^([^\?#]*)\??([^#]*)#?(\w*)$/i);
		if(!arr) return null;
		var r = {
		    'schema': '',
		    'host': '',
		    'path': arr[1],
		    'query': arr[2],
		    'anchor': arr[3],
		    'requesturi': arr[1] + (arr[2] ? '?'+arr[2] : '')  + (arr[3] ? '#'+arr[3] : '')
		};
		console.log(r);
		return r;
	}
}

【用例】 
var json = parse_url('http://bbs.xiuno.com/forum-1.htm?keyword=word#hash');
console.log(json);
