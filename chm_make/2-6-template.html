template(s, json)

批量字符串替换，一般用作模板变量替换。
【定义】 文件：xiunoui/xiuno.js 大约 313 行： 
global.template = function(s, json) {
	//console.log(json);
	for(k in json) {
		var r = new RegExp('\{('+k+')\}', 'g');
		s = s.replace(r, function(match, name) {
			return json[name];
		});
	}

	return s;
}

【用例】 
var s = "my name is {name}";
var json = {"name":"张无忌"}
var s2 = template(s, json);
console.log(s2);
// my name is 张无忌
