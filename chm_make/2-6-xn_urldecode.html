xn_urldecode(s)

解码 xn_urlencode() 生成的字符，返回字符串。
【定义】 文件：xiunoui/xiuno.js 大约 80 行： 
global.xn_urldecode = function(s) {
	s = s.replace('_', "%");
	s = decodeURIComponent(s);
	return s;
}

【用例】 
var s = "http_3a_2f_2fwww_2exiuno_2ecom_2f";
var s2 = xn_urldecode(s);
console.log(s2); 
// http://www.xiuno.com/
