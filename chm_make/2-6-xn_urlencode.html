xn_urlencode(s)

转义 URL 特殊字符，生成安全的可以通过 URL 传播的编码。
【定义】 文件：xiunoui/xiuno.js 大约 80 行： 
global.xn_urlencode = function(s) {
	s = encodeURIComponent(s);
	s = s.replace(/_/g, "%5f");
	s = s.replace(/\-/g, "%2d");
	s = s.replace(/\./g, "%2e");
	s = s.replace(/\~/g, "%7e");
	s = s.replace(/\!/g, "%21");
	s = s.replace(/\*/g, "%2a");
	s = s.replace(/\(/g, "%28");
	s = s.replace(/\)/g, "%29");
	s = s.replace(/\%/g, "_");
	return s;
}

【用例】 
var s = "http://www.xiuno.com/";
var s2 = xn_urlencode(s);
console.log(s2); 
// http_3a_2f_2fwww_2exiuno_2ecom_2f
